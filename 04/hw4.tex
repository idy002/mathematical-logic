	\begin{sol}{4.3} %
		
		It's not derivable.
		
		Prove by contradiction.
		
		Assume that it's derivable. Let $\Gamma$ be empty.
		
		Let $\fA$ be the S-structure of natural number. Let $\varphi$ be $x \equiv 1$.
		
		Then $\fA\models \exists x \varphi$, by our assumption, $\fA\models\forall x\varphi$. 
		But it's obvious that $\fA \not\models\forall x\varphi$. It's a contradiction.
		
		So it's not derivable.
		
	\end{sol}
